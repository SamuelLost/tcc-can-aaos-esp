236void MCP2515::Device::prepareId(uint8_t* buffer, const bool ext,
237 const uint32_t id) {
238 uint16_t can_id = static_cast<uint16_t>(id & 0x0FFFF);
239
240 if (ext) {
241 buffer[EID0] = static_cast<uint8_t>(can_id & 0xFF);
242 buffer[EID8] = static_cast<uint8_t>(can_id >> 8);
243 can_id = static_cast<uint16_t>(id >> 16);
244 buffer[SIDL] = static_cast<uint8_t>(can_id & 0x03);
245 buffer[SIDL] += static_cast<uint8_t>((can_id & 0x1C) << 3);
246 buffer[SIDL] |= TXB_EXIDE_MASK;
247 buffer[SIDH] = static_cast<uint8_t>(can_id >> 5);
248 } else {
249 buffer[SIDH] = static_cast<uint8_t>(can_id >> 3);
250 buffer[SIDL] = static_cast<uint8_t>((can_id & 0x07) << 5);
251 buffer[EID0] = 0;
252 buffer[EID8] = 0;
253 }
254}

389MCP2515::Error MCP2515::Device::readMessage(RXBn rxbn, CanMessage& message) {
390 const RXBn_REGS* rxb = &RXB[static_cast<int>(rxbn)];
391
392 uint8_t tbufdata[5];
393
394 readRegisters(rxb->SIDH, tbufdata, 5);
395
396 uint32_t id = (tbufdata[SIDH] << 3) + (tbufdata[SIDL] >> 5);
397
398 if ((tbufdata[SIDL] & TXB_EXIDE_MASK) == TXB_EXIDE_MASK) {
399 id = (id << 2) + (tbufdata[SIDL] & 0x03);
400 id = (id << 8) + tbufdata[EID8];
401 id = (id << 8) + tbufdata[EID0];
402 id |= CAN_EFF_FLAG;
403 }
404
405 uint8_t dlc = (tbufdata[DLC] & DLC_MASK);
406 if (dlc > CAN_MAX_DATA_LENGTH) {
407 return Error::FAIL;
408 }
409
410 uint8_t ctrl = readRegister(rxb->CTRL);
411 if (ctrl & RXBnCTRL_RTR) {
412 id |= CAN_RTR_FLAG;
413 }
414
415 message.identifier = id;
416 message.data_length_code = dlc;
417
418 readRegisters(rxb->DATA, message.data, dlc);
419
420 modifyRegister(Register::CANINTF, static_cast<int>(rxb->CANINTF_RXnIF), 0);
421
422 return Error::OK;
423}

425MCP2515::Error MCP2515::Device::sendMessage(CanMessage& message) {
426 if (message.data_length_code > CAN_MAX_DATA_LENGTH) {
427 return Error::FAIL_TX;
428 }
429
430 TXBn txBuffers[N_TXBUFFERS] = {TXBn::TXB0, TXBn::TXB1, TXBn::TXB2};
431
432 for (int i = 0; i < N_TXBUFFERS; i++) {
433 const TXBn_REGS* txbuf = &TXB[static_cast<int>(txBuffers[i])];
434 uint8_t ctrlval = readRegister(txbuf->CTRL);
435 if ((ctrlval & static_cast<int>(TXBnCTRL::TXREQ)) == 0) {
436 return sendMessage(txBuffers[i], message);
437 }
438 }
439
440 return Error::ALL_TX_BUSY;
441}

443MCP2515::Error MCP2515::Device::sendMessage(TXBn txbn, CanMessage& message) {
444 if (message.data_length_code > CAN_MAX_DATA_LENGTH) {
445 return Error::FAIL_TX;
446 }
447
448 const TXBn_REGS* txbuf = &TXB[static_cast<int>(txbn)];
449
450 uint8_t data[13];
451
452 bool ext = (message.identifier & CAN_EFF_FLAG);
453 bool rtr = (message.identifier & CAN_RTR_FLAG);
454 uint32_t id = (message.identifier & (ext ? CAN_EFF_MASK : CAN_SFF_MASK));
455
456 prepareId(data, ext, id);
457
458 data[DLC] =
459 rtr ? (message.data_length_code | RTR_MASK) : message.data_length_code;
460
461 memcpy(&data[DATA], message.data, message.data_length_code);
462
463 setRegisters(txbuf->SIDH, data, 5 + message.data_length_code);
464
465 modifyRegister(txbuf->CTRL, static_cast<int>(TXBnCTRL::TXREQ),
466 static_cast<int>(TXBnCTRL::TXREQ));
467
468 uint8_t ctrl = readRegister(txbuf->CTRL);
469 if ((ctrl &
470 (static_cast<int>(TXBnCTRL::ABTF) | static_cast<int>(TXBnCTRL::MLOA) |
471 static_cast<int>(TXBnCTRL::TXERR))) != 0) {
472 return Error::FAIL_TX;
473 }
474 return Error::OK;
475}
